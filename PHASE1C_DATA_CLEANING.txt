🔍 STEP 1: DUPLICATE DETECTION & REMOVAL
═══════════════════════════════════════════════════════════════════════════

WHY REMOVE DUPLICATES?
- Prevents data leakage (same image in train and validation)
- Improves model generalization
- Reduces storage requirements

METHOD: Perceptual Hashing (pHash)
- Compares image content, not pixel values
- Robust to minor transformations (resize, compression)
- Fast: O(n) average case

IMPLEMENTATION:
--------------

Install imagehash:
$ pip install imagehash Pillow

#!/usr/bin/env python3
"""
Remove duplicate images using perceptual hashing
"""

import imagehash
from PIL import Image
from pathlib import Path
from collections import defaultdict
import shutil
from tqdm import tqdm

def compute_hash(image_path, hash_size=8):
    """
    Compute perceptual hash of image
    
    Args:
        image_path: Path to image
        hash_size: Hash size (higher = more sensitive)
    
    Returns:
        imagehash object
    """
    try:
        img = Image.open(image_path)
        return imagehash.phash(img, hash_size=hash_size)
    except Exception as e:
        print(f"Error processing {image_path}: {e}")
        return None

def find_duplicates(image_dir, hash_size=8, threshold=5):
    """
    Find duplicate images in directory
    
    Args:
        image_dir: Directory containing images
        hash_size: Perceptual hash size
        threshold: Hamming distance threshold (0 = exact duplicates)
    
    Returns:
        dict: {original_image: [list of duplicates]}
    """
    image_paths = list(Path(image_dir).rglob("*.jpg")) + \
                  list(Path(image_dir).rglob("*.png"))
    
    print(f"Scanning {len(image_paths)} images...")
    
    # Compute hashes
    hashes = {}
    for img_path in tqdm(image_paths, desc="Computing hashes"):
        img_hash = compute_hash(img_path, hash_size)
        if img_hash:
            hashes[str(img_path)] = img_hash
    
    # Find duplicates by comparing hashes
    duplicates = defaultdict(list)
    processed = set()
    
    paths = list(hashes.keys())
    for i, path1 in enumerate(tqdm(paths, desc="Finding duplicates")):
        if path1 in processed:
            continue
        
        hash1 = hashes[path1]
        
        for path2 in paths[i+1:]:
            if path2 in processed:
                continue
            
            hash2 = hashes[path2]
            
            # Hamming distance between hashes
            distance = hash1 - hash2
            
            if distance <= threshold:
                duplicates[path1].append(path2)
                processed.add(path2)
    
    return duplicates

def remove_duplicates(duplicates, dry_run=True):
    """
    Remove duplicate images
    
    Args:
        duplicates: dict of {original: [duplicates]}
        dry_run: If True, only print what would be deleted
    """
    total_duplicates = sum(len(dups) for dups in duplicates.values())
    
    print(f"\nFound {len(duplicates)} groups with {total_duplicates} duplicates")
    
    if dry_run:
        print("\n[DRY RUN] Would delete:")
        for original, dups in list(duplicates.items())[:5]:
            print(f"  Keep: {original}")
            for dup in dups:
                print(f"    Delete: {dup}")
            print()
        print(f"... and {len(duplicates) - 5} more groups")
    else:
        for original, dups in tqdm(duplicates.items(), desc="Deleting duplicates"):
            for dup_path in dups:
                # Delete image
                Path(dup_path).unlink()
                
                # Delete corresponding label if exists
                label_path = Path(str(dup_path).replace('/images/', '/labels/').replace('.jpg', '.txt').replace('.png', '.txt'))
                if label_path.exists():
                    label_path.unlink()
        
        print(f"✅ Deleted {total_duplicates} duplicate images")

#!/usr/bin/env python3
"""
Filter low-quality images
"""

import cv2
import numpy as np
from pathlib import Path
from tqdm import tqdm
import shutil

def check_resolution(image_path, min_size=480):
    """Check if image meets minimum resolution"""
    try:
        img = cv2.imread(str(image_path))
        if img is None:
            return False, "Corrupted"
        
        h, w = img.shape[:2]
        if min(h, w) < min_size:
            return False, f"Low resolution: {w}×{h}"
        
        return True, "OK"
    except Exception as e:
        return False, f"Error: {e}"

def detect_blur(image_path, threshold=100):
    """
    Detect if image is blurry using Laplacian variance
    
    Args:
        threshold: Lower = more blurry (typical: 100-200)
    
    Returns:
        (is_sharp, variance)
    """
    try:
        img = cv2.imread(str(image_path))
        if img is None:
            return False, 0
        
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        laplacian_var = cv2.Laplacian(gray, cv2.CV_64F).var()
        
        return laplacian_var >= threshold, laplacian_var
    except Exception as e:
        return False, 0

def check_aspect_ratio(image_path, max_ratio=3.0):
    """Filter images with extreme aspect ratios"""
    try:
        img = cv2.imread(str(image_path))
        if img is None:
            return False, 0
        
        h, w = img.shape[:2]
        ratio = max(h, w) / min(h, w)
        
        return ratio <= max_ratio, ratio
    except Exception as e:
        return False, 0

def filter_images(input_dir, output_dir, 
                  min_resolution=480,
                  blur_threshold=100,
                  max_aspect_ratio=3.0):
    """
    Filter images by quality criteria
    
    Args:
        input_dir: Source directory
        output_dir: Destination for clean images
        min_resolution: Minimum dimension
        blur_threshold: Laplacian variance threshold
        max_aspect_ratio: Max width/height ratio
    """
    Path(output_dir).mkdir(parents=True, exist_ok=True)
    
    image_paths = list(Path(input_dir).rglob("*.jpg")) + \
                  list(Path(input_dir).rglob("*.png"))
    
    stats = {
        'total': len(image_paths),
        'passed': 0,
        'failed_resolution': 0,
        'failed_blur': 0,
        'failed_aspect': 0,
        'failed_corrupted': 0
    }
    
    for img_path in tqdm(image_paths, desc="Filtering images"):
        # Check resolution
        res_ok, res_msg = check_resolution(img_path, min_resolution)
        if not res_ok:
            if "Corrupted" in res_msg:
                stats['failed_corrupted'] += 1
            else:
                stats['failed_resolution'] += 1
            continue
        
        # Check blur
        sharp, blur_val = detect_blur(img_path, blur_threshold)
        if not sharp:
            stats['failed_blur'] += 1
            continue
        
        # Check aspect ratio
        aspect_ok, ratio = check_aspect_ratio(img_path, max_aspect_ratio)
        if not aspect_ok:
            stats['failed_aspect'] += 1
            continue
        
        # Passed all checks - copy to output
        rel_path = img_path.relative_to(input_dir)
        output_path = Path(output_dir) / rel_path
        output_path.parent.mkdir(parents=True, exist_ok=True)
        
        shutil.copy2(img_path, output_path)
        
        # Copy label if exists
        label_path = Path(str(img_path).replace('/images/', '/labels/').replace('.jpg', '.txt').replace('.png', '.txt'))
        if label_path.exists():
            output_label = Path(str(output_path).replace('/images/', '/labels/').replace('.jpg', '.txt').replace('.png', '.txt'))
            output_label.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(label_path, output_label)
        
        stats['passed'] += 1
    
    # Print statistics
    print("\n" + "="*50)
    print("IMAGE QUALITY FILTERING RESULTS")
    print("="*50)
    print(f"Total images scanned: {stats['total']}")
    print(f"✅ Passed: {stats['passed']} ({stats['passed']/stats['total']*100:.1f}%)")
    print(f"❌ Failed - Low resolution: {stats['failed_resolution']}")
    print(f"❌ Failed - Blurry: {stats['failed_blur']}")
    print(f"❌ Failed - Extreme aspect: {stats['failed_aspect']}")
    print(f"❌ Failed - Corrupted: {stats['failed_corrupted']}")
    print("="*50)
    
    # Save report
    with open('results/cleaning/quality_report.txt', 'w') as f:
        f.write("Image Quality Filtering Report\n")
        f.write("="*50 + "\n\n")
        for key, value in stats.items():
            f.write(f"{key}: {value}\n")
    
    return stats

if __name__ == "__main__":
    import argparse
    
    parser = argparse.ArgumentParser()
    parser.add_argument('--input_dir', default='dataset/raw')
    parser.add_argument('--output_dir', default='dataset/cleaned')
    parser.add_argument('--min_resolution', type=int, default=480)
    parser.add_argument('--blur_threshold', type=int, default=100)
    parser.add_argument('--max_aspect', type=float, default=3.0)

    )

#!/usr/bin/env python3
"""
Validate YOLO format annotations
"""

from pathlib import Path
import numpy as np
from tqdm import tqdm

def parse_yolo_label(label_path):
    """
    Parse YOLO format label file
    
    Returns:
        list of (class_id, x_center, y_center, width, height)
    """
    boxes = []
    with open(label_path) as f:
        for line in f:
            parts = line.strip().split()
            if len(parts) != 5:
                return None  # Invalid format
            
            try:
                class_id = int(parts[0])
                x, y, w, h = map(float, parts[1:])
                boxes.append((class_id, x, y, w, h))
            except ValueError:
                return None
    
    return boxes

def validate_box(box, num_classes=8):
    """Validate single bounding box"""
    class_id, x, y, w, h = box
    
    issues = []
    
    # Check class ID
    if class_id < 0 or class_id >= num_classes:
        issues.append(f"Invalid class ID: {class_id}")
    
    # Check coordinates in [0, 1]
    if not (0 <= x <= 1 and 0 <= y <= 1):
        issues.append(f"Coordinates out of range: ({x:.3f}, {y:.3f})")
    
    # Check dimensions
    if w <= 0 or h <= 0:
        issues.append(f"Invalid dimensions: {w:.3f} × {h:.3f}")
    
    if w > 1 or h > 1:
        issues.append(f"Dimensions > 1: {w:.3f} × {h:.3f}")
    
    # Check aspect ratio
    if w > 0 and h > 0:
        aspect = max(w, h) / min(w, h)
        if aspect > 10:
            issues.append(f"Extreme aspect ratio: {aspect:.1f}:1")
    
    # Check if box extends beyond image
    x_min = x - w/2
    x_max = x + w/2
    y_min = y - h/2
    y_max = y + h/2
    
    if x_min < 0 or x_max > 1 or y_min < 0 or y_max > 1:
        issues.append(f"Box extends beyond image: [{x_min:.3f}, {y_min:.3f}, {x_max:.3f}, {y_max:.3f}]")
    
    return issues

def validate_annotations(dataset_dir, num_classes=8):
    """
    Validate all annotations in dataset
    
    Returns:
        dict: Statistics and invalid files
    """
    label_paths = list(Path(dataset_dir).rglob("*.txt"))
    
    stats = {
        'total_files': len(label_paths),
        'valid_files': 0,
        'invalid_files': 0,
        'total_boxes': 0,
        'invalid_boxes': 0,
        'issues': {}
    }
    
    invalid_files = []
    
    for label_path in tqdm(label_paths, desc="Validating annotations"):
        boxes = parse_yolo_label(label_path)
        
        if boxes is None:
            stats['invalid_files'] += 1
            invalid_files.append((str(label_path), "Invalid format"))
            continue
        
        file_valid = True
        for box in boxes:
            stats['total_boxes'] += 1
            issues = validate_box(box, num_classes)
            
            if issues:
                stats['invalid_boxes'] += 1
                file_valid = False
                for issue in issues:
                    stats['issues'][issue] = stats['issues'].get(issue, 0) + 1
                invalid_files.append((str(label_path), issues))
        
        if file_valid:
            stats['valid_files'] += 1
        else:
            stats['invalid_files'] += 1
    
    return stats, invalid_files

def fix_annotations(invalid_files, auto_fix=False):
    """
    Attempt to fix common annotation issues
    
    Args:
        invalid_files: List of (path, issues)
        auto_fix: If True, apply automatic fixes
    """
    fixable_count = 0
    
    for file_path, issues in invalid_files:
        # Example fixes:
        # 1. Clip coordinates to [0, 1]
        # 2. Remove boxes with invalid dimensions
        # 3. Correct class IDs if mapping exists
        
        if auto_fix:
            # Implementation of automatic fixes
            pass
        else:
            print(f"Would fix: {file_path}")
            print(f"  Issues: {issues}\n")
            fixable_count += 1
    
    return fixable_count

#!/usr/bin/env python3
"""
Standardize labels across multiple dataset sources
"""

import yaml
from pathlib import Path
import shutil
from tqdm import tqdm

# Standard class mapping
STANDARD_CLASSES = {
    0: 'butterflyfish',
    1: 'grouper',
    2: 'parrotfish',
    3: 'surgeonfish',
    4: 'damselfish',
    5: 'wrasse',
    6: 'triggerfish',
    7: 'angelfish'
}

def load_species_mapping(config_path='configs/species_mapping.yaml'):
    """Load species to family mapping from YAML"""
    with open(config_path) as f:
        data = yaml.safe_load(f)
    
    # Create reverse mapping: species_name -> family_index
    reverse_map = {}
    for family, info in data['indicator_families'].items():
        family_id = list(STANDARD_CLASSES.values()).index(family)
        for species in info['species']:
            species_norm = species.lower().strip()
            reverse_map[species_norm] = family_id
    
    return reverse_map

def convert_dataset(input_dir, output_dir, dataset_classes, species_mapping):
    """
    Convert dataset to standard class IDs
    
    Args:
        input_dir: Source dataset directory
        output_dir: Output directory
        dataset_classes: Original class names in dataset
        species_mapping: Species -> family mapping
    """
    # Create class ID mapping: old_id -> new_id
    class_mapping = {}
    unmapped_classes = []
    
    for old_id, class_name in enumerate(dataset_classes):
        class_norm = class_name.lower().strip()
        
        # Try exact match
        if class_norm in STANDARD_CLASSES.values():
            new_id = list(STANDARD_CLASSES.values()).index(class_norm)
            class_mapping[old_id] = new_id
        # Try species mapping
        elif class_norm in species_mapping:
            new_id = species_mapping[class_norm]
            class_mapping[old_id] = new_id
        else:
            unmapped_classes.append(class_name)
    
    if unmapped_classes:
        print(f"⚠️ Unmapped classes: {unmapped_classes}")
        print("   These will be skipped.")
    
    # Convert label files
    label_paths = list(Path(input_dir).rglob("*.txt"))
    
    for label_path in tqdm(label_paths, desc=f"Converting {input_dir.name}"):
        # Read original labels
        converted_lines = []
        with open(label_path) as f:
            for line in f:
                parts = line.strip().split()
                if len(parts) != 5:
                    continue
                
                old_class_id = int(parts[0])
                
                # Skip if class not mapped
                if old_class_id not in class_mapping:
                    continue
                
                # Convert to new class ID
                new_class_id = class_mapping[old_class_id]
                converted_lines.append(f"{new_class_id} {' '.join(parts[1:])}\n")
        
        # Save converted labels
        if converted_lines:
            rel_path = label_path.relative_to(input_dir)
            output_path = Path(output_dir) / rel_path
            output_path.parent.mkdir(parents=True, exist_ok=True)
            
            with open(output_path, 'w') as f:
                f.writelines(converted_lines)
            
            # Copy corresponding image
            img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.jpg'))
            if not img_path.exists():
                img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.png'))
            
            if img_path.exists():
                output_img = Path(str(output_path).replace('/labels/', '/images/').replace('.txt', img_path.suffix))
                output_img.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(img_path, output_img)

Generate comprehensive dataset statistics
