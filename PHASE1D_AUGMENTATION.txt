#!/usr/bin/env python3
"""
Analyze class balance and calculate augmentation needs
"""

from pathlib import Path
from collections import defaultdict

def analyze_dataset(dataset_dir, target_per_class=800):
    """Calculate how many augmented images needed per class"""
    
    # Count current distribution
    class_counts = defaultdict(int)
    image_paths_by_class = defaultdict(list)
    
    label_paths = list(Path(dataset_dir).rglob("*.txt"))
    
    for label_path in label_paths:
        with open(label_path) as f:
            classes_in_image = set()
            for line in f:
                parts = line.strip().split()
                if len(parts) == 5:
                    class_id = int(parts[0])
                    classes_in_image.add(class_id)
            
            # Count image for each class present
            for class_id in classes_in_image:
                class_counts[class_id] += 1
                
                # Get corresponding image path
                img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.jpg'))
                if not img_path.exists():
                    img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.png'))
                if img_path.exists():
                    image_paths_by_class[class_id].append(img_path)
    
    # Calculate augmentation needs
    class_names = {
        0: 'Butterflyfish', 1: 'Grouper', 2: 'Parrotfish', 3: 'Surgeonfish',
        4: 'Damselfish', 5: 'Wrasse', 6: 'Triggerfish', 7: 'Angelfish'
    }
    
    augmentation_plan = {}
    
    print("\n" + "="*70)
    print("CLASS BALANCE ANALYSIS")
    print("="*70)
    
    for class_id in sorted(class_counts.keys()):
        current = class_counts[class_id]
        needed = max(0, target_per_class - current)
        aug_factor = (target_per_class / current) if current > 0 else 0
        
        augmentation_plan[class_id] = {
            'name': class_names[class_id],
            'current': current,
            'target': target_per_class,
            'needed': needed,
            'augmentation_factor': aug_factor,
            'image_paths': image_paths_by_class[class_id]
        }
        
        print(f"{class_names[class_id]:15s}: {current:4d} → {target_per_class:4d} "
              f"(+{needed:3d} needed, {aug_factor:.2f}x)")
    
    total_current = sum(class_counts.values())
    total_target = target_per_class * len(class_counts)
    
    print("="*70)
    print(f"TOTAL: {total_current:4d} → {total_target:4d} "
          f"(+{total_target - total_current} needed)")
    
    balance_ratio_before = max(class_counts.values()) / min(class_counts.values())
    balance_ratio_after = 1.0  # Perfect balance
    
    print(f"\nBalance ratio: {balance_ratio_before:.1f}:1 → {balance_ratio_after:.1f}:1")
    print("="*70)
    
    return augmentation_plan

if __name__ == "__main__":
    plan = analyze_dataset('dataset/standardized', target_per_class=800)
    
    # Save plan
    import json
    with open('configs/augmentation_plan.json', 'w') as f:
        # Remove image_paths from JSON (too large)
        plan_json = {k: {kk: vv for kk, vv in v.items() if kk != 'image_paths'} 
                     for k, v in plan.items()}
        json.dump(plan_json, f, indent=2)
    
    print(f"\n✅ Augmentation plan saved to configs/augmentation_plan.json")

SCIENTIFIC JUSTIFICATION:
------------------------
All augmentations must reflect real-world underwater conditions:

1. HSV COLOR SHIFT:
   - Simulates different water depths (deeper = more blue/green shift)
   - Validates against Khiem et al. (PLOS ONE 2025)
   - Parameters: H(±0.02), S(±0.7), V(±0.4)

2. HORIZONTAL FLIP:
   - Fish swim in all directions
   - 50% probability
   - Maintains biological validity

3. ROTATION:
   - Fish can be oriented at any angle
   - ±15° (avoid extreme tilts that look unnatural)

4. BRIGHTNESS/CONTRAST:
   - Simulates different lighting conditions (dawn/midday/dusk)
   - ±20% adjustment

5. GAUSSIAN NOISE:
   - Simulates water turbidity
   - σ=5 (subtle)

AVOID THESE (NOT ECOLOGICALLY VALID):
- Vertical flip (fish don't swim upside-down normally)
- Extreme rotation (>30°)
- Grayscale conversion (color is important for species ID)
- Cutout/random erasing (creates unrealistic occlusions)

#!/usr/bin/env python3
"""
Offline augmentation for class balancing
Applies only to under-represented classes
"""

import cv2
import numpy as np
from pathlib import Path
import random
from tqdm import tqdm
import shutil

def hsv_augment(image, h_gain=0.02, s_gain=0.7, v_gain=0.4):
    """
    HSV color space augmentation
    Simulates depth variations in underwater images
    """
    r = np.random.uniform(-1, 1, 3) * [h_gain, s_gain, v_gain] + 1
    hue, sat, val = cv2.split(cv2.cvtColor(image, cv2.COLOR_BGR2HSV))
    
    dtype = image.dtype
    x = np.arange(0, 256, dtype=np.int16)
    
    lut_hue = ((x * r[0]) % 180).astype(dtype)
    lut_sat = np.clip(x * r[1], 0, 255).astype(dtype)
    lut_val = np.clip(x * r[2], 0, 255).astype(dtype)
    
    img_hsv = cv2.merge((cv2.LUT(hue, lut_hue), 
                         cv2.LUT(sat, lut_sat), 
                         cv2.LUT(val, lut_val)))
    
    return cv2.cvtColor(img_hsv, cv2.COLOR_HSV2BGR)

def brightness_contrast_augment(image, brightness=0.2, contrast=0.2):
    """Adjust brightness and contrast"""
    alpha = 1.0 + random.uniform(-contrast, contrast)  # Contrast
    beta = random.uniform(-brightness, brightness) * 255  # Brightness
    
    return cv2.convertScaleAbs(image, alpha=alpha, beta=beta)

def gaussian_noise_augment(image, sigma=5):
    """Add Gaussian noise to simulate turbidity"""
    noise = np.random.normal(0, sigma, image.shape).astype(np.float32)
    noisy_image = image.astype(np.float32) + noise
    return np.clip(noisy_image, 0, 255).astype(np.uint8)

def rotate_image_and_boxes(image, boxes, angle):
    """
    Rotate image and update bounding boxes
    
    Args:
        image: Input image
        boxes: List of (class_id, x_center, y_center, width, height) in YOLO format
        angle: Rotation angle in degrees
    
    Returns:
        rotated_image, rotated_boxes
    """
    h, w = image.shape[:2]
    center = (w // 2, h // 2)
    
    # Rotation matrix
    M = cv2.getRotationMatrix2D(center, angle, 1.0)
    
    # Rotate image
    rotated = cv2.warpAffine(image, M, (w, h), 
                             flags=cv2.INTER_LINEAR,
                             borderMode=cv2.BORDER_CONSTANT,
                             borderValue=(0, 0, 0))
    
    # Rotate boxes
    rotated_boxes = []
    for class_id, x_c, y_c, box_w, box_h in boxes:
        # Convert to pixel coordinates
        x_center_px = x_c * w
        y_center_px = y_c * h
        
        # Rotate center point
        coords = np.array([[x_center_px, y_center_px]])
        rotated_coords = cv2.transform(coords.reshape(1, -1, 2), M).reshape(-1, 2)
        
        # Convert back to normalized
        x_new = rotated_coords[0, 0] / w
        y_new = rotated_coords[0, 1] / h
        
        # Keep within bounds
        x_new = np.clip(x_new, 0, 1)
        y_new = np.clip(y_new, 0, 1)
        
        rotated_boxes.append((class_id, x_new, y_new, box_w, box_h))
    
    return rotated, rotated_boxes

def flip_horizontal(image, boxes):
    """Horizontal flip with box adjustment"""
    flipped_image = cv2.flip(image, 1)
    
    flipped_boxes = []
    for class_id, x_c, y_c, w, h in boxes:
        # Flip x-coordinate
        x_new = 1.0 - x_c
        flipped_boxes.append((class_id, x_new, y_c, w, h))
    
    return flipped_image, flipped_boxes

def augment_image(image_path, label_path, output_dir, aug_id, techniques):
    """
    Apply augmentation techniques and save
    
    Args:
        image_path: Input image
        label_path: Input label
        output_dir: Where to save augmented data
        aug_id: Augmentation identifier
        techniques: List of techniques to apply
    """
    # Read image
    image = cv2.imread(str(image_path))
    if image is None:
        return False
    
    # Read boxes
    boxes = []
    with open(label_path) as f:
        for line in f:
            parts = line.strip().split()
            if len(parts) == 5:
                class_id = int(parts[0])
                coords = list(map(float, parts[1:]))
                boxes.append((class_id, *coords))
    
    # Apply techniques
    for technique in techniques:
        if technique == 'hsv':
            image = hsv_augment(image)
        elif technique == 'brightness':
            image = brightness_contrast_augment(image)
        elif technique == 'noise':
            image = gaussian_noise_augment(image)
        elif technique == 'hflip':
            image, boxes = flip_horizontal(image, boxes)
        elif technique.startswith('rotate'):
            angle = int(technique.split('_')[1])
            image, boxes = rotate_image_and_boxes(image, boxes, angle)
    
    # Save augmented image
    img_name = image_path.stem + f"_aug{aug_id}{image_path.suffix}"
    output_img_path = output_dir / "images" / img_name
    output_img_path.parent.mkdir(parents=True, exist_ok=True)
    cv2.imwrite(str(output_img_path), image)
    
    # Save augmented labels
    label_name = image_path.stem + f"_aug{aug_id}.txt"
    output_label_path = output_dir / "labels" / label_name
    output_label_path.parent.mkdir(parents=True, exist_ok=True)
    
    with open(output_label_path, 'w') as f:
        for class_id, x, y, w, h in boxes:
            f.write(f"{class_id} {x:.6f} {y:.6f} {w:.6f} {h:.6f}\n")
    
    return True

def augment_class(class_images, class_id, current_count, target_count, output_dir):
    """
    Augment specific class to reach target count
    
    Args:
        class_images: List of image paths for this class
        class_id: Class identifier
        current_count: Current number of images
        target_count: Target number of images
        output_dir: Output directory
    """
    if current_count >= target_count:
        # No augmentation needed - just copy originals
        print(f"  Class {class_id}: No augmentation needed ({current_count} >= {target_count})")
        return
    
    needed = target_count - current_count
    aug_per_image = needed / len(class_images)
    
    print(f"  Class {class_id}: Generating {needed} augmented images ({aug_per_image:.1f} per original)")
    
    # Define augmentation strategies
    strategies = [
        ['hsv'],
        ['brightness'],
        ['hflip'],
        ['hsv', 'brightness'],
        ['hsv', 'hflip'],
        ['rotate_10'],
        ['rotate_-10'],
        ['rotate_15'],
        ['hsv', 'noise'],
        ['brightness', 'hflip'],
    ]
    
    aug_count = 0
    aug_id = 0
    
    # Cycle through images and apply different augmentations
    while aug_count < needed:
        for img_path in class_images:
            if aug_count >= needed:
                break
            
            # Get corresponding label
            label_path = Path(str(img_path).replace('/images/', '/labels/').replace('.jpg', '.txt').replace('.png', '.txt'))
            
            if not label_path.exists():
                continue
            
            # Select random strategy
            strategy = random.choice(strategies)
            
            # Apply augmentation
            success = augment_image(img_path, label_path, output_dir, aug_id, strategy)
            
            if success:
                aug_count += 1
                aug_id += 1

def create_balanced_dataset(input_dir, output_dir, target_per_class=800):
    """
    Create balanced dataset through selective augmentation
    
    Args:
        input_dir: Cleaned dataset directory
        output_dir: Output directory for balanced dataset
        target_per_class: Target number of images per class
    """
    # First, analyze current distribution
    from analyze_balance import analyze_dataset
    augmentation_plan = analyze_dataset(input_dir, target_per_class)
    
    # Create output directories
    (Path(output_dir) / "images").mkdir(parents=True, exist_ok=True)
    (Path(output_dir) / "labels").mkdir(parents=True, exist_ok=True)
    
    print("\n" + "="*70)
    print("AUGMENTATION PROCESS")
    print("="*70)
    
    # Copy all original images first
    print("\n1. Copying original images...")
    label_paths = list(Path(input_dir).rglob("*.txt"))
    
    for label_path in tqdm(label_paths, desc="Copying originals"):
        # Copy label
        rel_path = label_path.relative_to(input_dir)
        output_label = Path(output_dir) / rel_path
        output_label.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(label_path, output_label)
        
        # Copy image
        img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.jpg'))
        if not img_path.exists():
            img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.png'))
        
        if img_path.exists():
            rel_img = img_path.relative_to(input_dir)
            output_img = Path(output_dir) / rel_img
            output_img.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(img_path, output_img)
    
    # Augment under-represented classes
    print("\n2. Augmenting under-represented classes...")
    for class_id, plan in augmentation_plan.items():
        if plan['needed'] > 0:
            augment_class(
                plan['image_paths'],
                class_id,
                plan['current'],
                plan['target'],
                Path(output_dir)
            )
    
    print("\n" + "="*70)
    print("✅ Augmentation complete!")
    print("="*70)

if __name__ == "__main__":
    import argparse
    
    parser = argparse.ArgumentParser()
    parser.add_argument('--input_dir', default='dataset/standardized')
    parser.add_argument('--output_dir', default='dataset/augmented')
    parser.add_argument('--target_per_class', type=int, default=800)
    args = parser.parse_args()
    
    create_balanced_dataset(args.input_dir, args.output_dir, args.target_per_class)

#!/usr/bin/env python3
"""
Split dataset into train/val/test with stratification
"""

from pathlib import Path
from sklearn.model_selection import train_test_split
from collections import defaultdict
import shutil
from tqdm import tqdm
import random

def get_image_class(label_path):
    """Get primary class from label file"""
    with open(label_path) as f:
        for line in f:
            parts = line.strip().split()
            if len(parts) == 5:
                return int(parts[0])  # Return first class found
    return None

def split_dataset(input_dir, output_dir, 
                  train_ratio=0.85, val_ratio=0.08, test_ratio=0.07):
    """
    Split dataset with stratification by class
    
    Args:
        input_dir: Augmented dataset directory
        output_dir: Output directory (will create train/val/test subdirs)
        train_ratio, val_ratio, test_ratio: Split ratios (must sum to 1.0)
    """
    assert abs(train_ratio + val_ratio + test_ratio - 1.0) < 0.001
    
    # Collect all label files
    label_paths = list(Path(input_dir).rglob("*.txt"))
    
    # Group by class for stratification
    class_files = defaultdict(list)
    for label_path in label_paths:
        class_id = get_image_class(label_path)
        if class_id is not None:
            class_files[class_id].append(label_path)
    
    # Split each class separately (stratified)
    train_files = []
    val_files = []
    test_files = []
    
    random.seed(42)  # Reproducibility
    
    for class_id, files in class_files.items():
        random.shuffle(files)
        
        n = len(files)
        n_train = int(n * train_ratio)
        n_val = int(n * val_ratio)
        
        train_files.extend(files[:n_train])
        val_files.extend(files[n_train:n_train+n_val])
        test_files.extend(files[n_train+n_val:])
    
    # Create split directories
    splits = {
        'train': train_files,
        'val': val_files,
        'test': test_files
    }
    
    print("\n" + "="*70)
    print("DATASET SPLITTING")
    print("="*70)
    print(f"Total images: {len(label_paths)}")
    print(f"Train: {len(train_files)} ({len(train_files)/len(label_paths)*100:.1f}%)")
    print(f"Val:   {len(val_files)} ({len(val_files)/len(label_paths)*100:.1f}%)")
    print(f"Test:  {len(test_files)} ({len(test_files)/len(label_paths)*100:.1f}%)")
    print("="*70)
    
    # Copy files to split directories
    for split_name, files in splits.items():
        split_dir = Path(output_dir) / split_name
        (split_dir / "images").mkdir(parents=True, exist_ok=True)
        (split_dir / "labels").mkdir(parents=True, exist_ok=True)
        
        print(f"\nCopying {split_name} split...")
        for label_path in tqdm(files):
            # Copy label
            label_dst = split_dir / "labels" / label_path.name
            shutil.copy2(label_path, label_dst)
            
            # Copy image
            img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.jpg'))
            if not img_path.exists():
                img_path = Path(str(label_path).replace('/labels/', '/images/').replace('.txt', '.png'))
            
            if img_path.exists():
                img_dst = split_dir / "images" / img_path.name
                shutil.copy2(img_path, img_dst)
    
    print("\n✅ Dataset split complete!")
    
    # Print class distribution per split
    print("\nClass distribution per split:")
    for split_name, files in splits.items():
        class_counts = defaultdict(int)
        for label_path in files:
            class_id = get_image_class(label_path)
            if class_id is not None:
                class_counts[class_id] += 1
        
        print(f"\n{split_name.upper()}:")
        for class_id in sorted(class_counts.keys()):
            print(f"  Class {class_id}: {class_counts[class_id]}")

if __name__ == "__main__":
    import argparse
    
    parser = argparse.ArgumentParser()
    parser.add_argument('--input_dir', default='dataset/augmented')
    parser.add_argument('--output_dir', default='dataset/processed')
    args = parser.parse_args()
    
    split_dataset(args.input_dir, args.output_dir)
